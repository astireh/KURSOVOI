ВВЕДЕНИЕ

Анализ и исследование математических функций являются фундаментальными задачами в вычислительной математике, физике, инженерии и экономике. В процессе обучения и практической работы часто возникает необходимость в получении конкретных численных результатов: вычислении значений, построении таблиц, интегрировании, решении уравнений и дифференцировании. 2.2 Анализ кода программы

В данной курсовой работе разработана программа на языке C, предназначенная для исследования и анализа кусочно-заданной математической функции. Программа предоставляет интерактивный интерфейс для выполнения различных математических операций, включая вычисление значений функции в точках, построение таблиц значений, численное интегрирование, поиск корней уравнений и вычисление производных.
В начале программы подключаются стандартные библиотеки: stdio.h отвечает за ввод и вывод — например, когда пользователь вводит числа или программа показывает результаты. Библиотека math.h содержит математические функции вроде синуса, экспоненты и возведения в степень. Библиотека locale.h помогает правильно отображать русский текст.
В программе есть две важные константы. EPS задаёт точность вычислений — она равна 0.000001, и это нужно для сравнения чисел с плавающей точкой. MAX_ITER ограничивает количество повторений в циклах, чтобы программа не зациклилась.

Программа работает в цикле — после выполнения каждого действия она снова показывает меню, пока пользователь не выберет выход. Сама функция определена тремя разными способами в зависимости от X:

Для x < -0.5: sin(2*x)/x использует тригонометрическую функцию синуса, деление на ноль при x = 0, но эта точка не входит в данный интервал.
Для -0.5 ≤ x < 0: exp(1/x) использует экспоненциальную функцию, при x - 0⁻ значение стремится к 0 (поскольку 1/x → -∞), при x - -∞ значение стремится к 1.

Функция build_table строит таблицу. Пользователь вводит начало и конец отрезка, а также сколько точек нужно рассчитать. Программа равномерно распределяет точки на отрезке, вычисляет в каждой из них значение функции и красиво выводит таблицу.
Функция calculate_integral вычисляет интеграл методом трапеций. Пользователь задаёт границы интегрирования и на сколько частей разбить отрезок. Программа вычисляет площадь под кривой как сумму площадей маленьких трапеций.
Функция find_x_by_y решает уравнение f(x) = y. Пользователь вводит значение Y, а программа ищет, при каких X функция принимает это значение. Сначала она ищет интервалы, где функция пересекает нужный уровень, а потом уточняет корень методом деления пополам.
Функция calculate_derivative вычисляет производную. Она использует метод центральных разностей — вычисляет функцию в точках x+h и x-h, а потом находит производную по формуле. Шаг h выбран небольшим (0.0001), чтобы обеспечить хорошую точность.
Программа проверяет вводимые данные на корректность. Например, если пользователь пытается построить таблицу с отрицательным количеством шагов или вычислить интеграл, где начало больше конца, программа сообщит об ошибке.
Все вычисления выполняются с высокой точностью, используются числа типа double. Результаты выводятся с разумным количеством знаков после запятой, чтобы их было удобно читать.

#define _CRT_SECURE_NO_WARNINGS

#define EPS 1e-6
#define MAX_ITER 100

void calculate_derivative(void);

int main(void)
{
    setlocale(LC_ALL, "RUS");
    int choice;

    do
    {
        printf("\n--- МЕНЮ ---\n");
        printf("1. Значение в точке\n");
        printf("2. Таблица значений\n");
        printf("3. Интеграл (метод трапеций)\n");

        printf("0. Выход\n");
        printf("Выберите: ");
        scanf("%d", &choice);

        switch (choice)
        {
        case 1: calculate_value(); break;
        case 2: build_table(); break;
        case 3: calculate_integral(); break;
        case 4: find_x_by_y(); break;
        case 5: calculate_derivative(); break;
        case 0: printf("Выход.\n"); break;
        default: printf("Неверный выбор!\n");
        }
    } while (choice != 0);
    return 0;
}

double f(double x)
{
    if (x < -0.5)
        return sin(2 * x) / x;
    else if (x < 0)
        return exp(1 / x);
    else
    {
        double sum = 0, fact = 1;
        for (int n = 0; n <= 4; n++)
        {
            if (n > 0) fact *= n;
            sum += pow(-1, n) * pow(x, 2 * n) / sqrt(fact);
        }
        return sum;
    }
}

void calculate_value(void)
{
    double x;
    printf("\nВведите x: ");
    scanf("%lf", &x);
    printf("f(%.4f) = %.6f\n", x, f(x));
}

void build_table(void)
{
    double start, end, step;
    int steps;

    printf("\nНачало интервала: ");
    scanf("%lf", &start);
    printf("Конец интервала: ");
    scanf("%lf", &end);
    printf("Количество шагов: ");
    scanf("%d", &steps);

    if (steps <= 0) { printf("Ошибка!\n"); return; }

    step = (end - start) / steps;
    printf("\n|     x     |    f(x)    |\n");
    printf("|-----------|------------|\n");

    for (int i = 0; i <= steps; i++)
    {
        double x = start + i * step;
        printf("| %9.4f | %10.6f |\n", x, f(x));
    }
}

void calculate_integral(void)
{
    double a, b, sum = 0, step;
    int n;

    printf("\nНачало интервала: ");
    scanf("%lf", &a);
    printf("Конец интервала: ");
    scanf("%lf", &b);
    printf("Количество разбиений: ");
    scanf("%d", &n);

    if (n <= 0 || a >= b) { printf("Ошибка!\n"); return; }

    step = (b - a) / n;

    for (int i = 0; i <= n; i++)
    {
        double x = a + i * step;
        double y = f(x);
        if (i == 0 || i == n)
            sum += y / 2;
        else
            sum += y;
    }

    sum *= step;
    printf("f(x)dx = %.6f на [%.2f, %.2f]\n", sum, a, b);
}

void find_x_by_y(void)
{
    double y, x, dx = 0.1;
    int found = 0;

    printf("\nВведите Y: ");
    scanf("%lf", &y);

    for (x = -10; x <= 10; x += dx)
    {
        double f1 = f(x) - y;
        double f2 = f(x + dx) - y;

        if (f1 * f2 <= 0)
        {
            double a = x, b = x + dx;
            for (int i = 0; i < MAX_ITER; i++)
            {
                double mid = (a + b) / 2;
                double fmid = f(mid) - y;

                if (fabs(fmid) < EPS) { x = mid; break; }
                if ((f(a) - y) * fmid < 0) b = mid;
                else a = mid;
            }
            printf("Найден x = %.6f (f(x) = %.6f)\n", x, f(x));
            found = 1;
        }
    }

    if (!found) printf("Решений не найдено на [-10, 10]\n");
}

void calculate_derivative(void)
{
    double x, h = 0.0001;

    printf("\nВведите x: ");
    scanf("%lf", &x);

    double deriv = (f(x + h) - f(x - h)) / (2 * h);
    printf("f'(%.4f) = %.6f\n", x, deriv);
}
